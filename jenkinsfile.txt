pipeline {
    agent any

    environment {
        SONARQUBE_SERVER = 'sonarqube' // The name configured in Jenkins
    }

    stages {
        
        stage('Build') {
            steps {
                echo 'Building the project...'
                
                // Run Maven clean install
                sh 'mvn clean install'
            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
                
                // Run tests, they are usually included in the 'mvn clean install' process
                sh 'mvn test'
            }
        }

        stage('SonarQube Analysis') {
            steps {
                echo 'Running SonarQube analysis...'
                
                // Run SonarQube scanner using Maven
                withSonarQubeEnv("${SONARQUBE_SERVER}") {
                    sh """
mvn sonar:sonar \
    -Dsonar.login=admin \
    -Dsonar.password=Yassine@1234
			"""
                  }
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
