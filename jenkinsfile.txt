pipeline {
    agent any

    environment {
registry = "getghosted/event"
registryCredential = '777'
dockerImage = 'event-app'
    }

    stages {
        stage('Build') {
            steps {
                echo 'Building the project...'
                
                // Run Maven to build and generate the binary
                sh 'mvn clean install'
            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
                
                // Run unit tests
                sh 'mvn test'
            }
        }

        stage('Building our image') {
          steps{
            script {
               dockerImage = docker.build registry + ":$BUILD_NUMBER"
}
}
}
        stage('Deploy our image') {
          steps{
            script {
              docker.withRegistry( '', registryCredential ) {
              dockerImage.push()
}
}
}
}
           stage('Cleaning up') {
             steps{
             sh "docker rmi $registry:$BUILD_NUMBER"
}
}
}
}

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
